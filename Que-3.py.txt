# Function to calculate grade based on average marks
def calculate_grade(avg_marks):
    if avg_marks >= 90:
        return "A+"
    elif avg_marks >= 80:
        return "A"
    elif avg_marks >= 70:
        return "B"
    elif avg_marks >= 60:
        return "C"
    elif avg_marks >= 50:
        return "D"
    else:
        return "F"

# Taking marks input for 5 subjects
marks = []
for i in range(1, 6):
    subject_marks = float(input(f"Enter marks for Subject {i}: "))
    marks.append(subject_marks)

# Calculating total marks
total_marks = sum(marks)

# Calculating average marks
average_marks = total_marks / 5

# Calculating grade based on average marks
grade = calculate_grade(average_marks)

# Printing the marksheet
print("\n----------- Marksheet -----------")
for i in range(5):
    print(f"Subject {i+1}: {marks[i]}")
print(f"Total Marks: {total_marks}")
print(f"Average Marks: {average_marks}")
print(f"Grade: {grade}")
